       package main
       
       import "fmt"

       func Least(x []int) int {
           r := 10000
           for i := range x {
               if x[i] < r {
                   r = x[i]
               }
           }
           return r
       }

       func main() {
	       x := []int{48,96,86,68,57,82,63,70,37,34,83,27,19,97,9,17,}
